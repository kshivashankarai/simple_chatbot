CREATE TABLE "users" (
	"id" VARCHAR NULL DEFAULT NULL,
	"code" VARCHAR(15) NULL DEFAULT NULL,
	"first_name" VARCHAR(20) NULL DEFAULT NULL,
	"middle_name" VARCHAR(20) NULL DEFAULT NULL,
	"last_name" VARCHAR(20) NULL DEFAULT NULL,
	"email" VARCHAR(55) NULL DEFAULT NULL,
	"phone_number" VARCHAR(13) NULL DEFAULT NULL,
	"date_of_birth" DATE NULL DEFAULT NULL,
	"department_id" INTEGER NULL DEFAULT NULL,
	"is_active" BOOLEAN NULL DEFAULT NULL
)
;


CREATE TABLE "department" (
	"id" SERIAL NOT NULL,
	"name" VARCHAR(15) NULL DEFAULT NULL
)
;


CREATE TABLE "roles" (
	"id" SERIAL NOT NULL,
	"name" VARCHAR(15) NULL DEFAULT NULL
)
;
